function updateCard(id, data) { // 增加更嚴謹的檢查，避免因為個別 API 失敗導致網頁卡死 if (!data || data.error || !data.chart?.result) { addLog(`⚠️ ${id} 資料獲取失敗`); return null; } const res = data.chart.result[0]; const m = res.meta; const price = m.regularMarketPrice; const prev = m.previousClose; const diff = price - prev; const perc = (diff / prev) * 100; // 重新計算百分比 const color = diff >= 0 ? '#ff4d4d' : '#2ecc71'; // 更新價格與百分比內容 document.getElementById(`p-${id}`).innerText = price.toLocaleString(undefined, {minimumFractionDigits: (id==='7769'||id==='twii')?0:2}); document.getElementById(`p-${id}`).style.color = color; // 顯示漲跌點數與百分比 document.getElementById(`diff-${id}`).innerText = `${diff >= 0 ? '▲' : '▼'} ${Math.abs(diff).toFixed(2)} (${perc.toFixed(2)}%)`; document.getElementById(`diff-${id}`).style.color = color; // 更新卡片顏色狀態 document.getElementById(`card-${id}`).className = `card ${id==='gold'?'full-width':''} ${diff >= 0 ? 'up' : 'down'}`; // 渲染曲線圖 try { const h = res.indicators.quote[0].close || []; const vh = h.filter(v => v !== null); if (vh.length > 0) renderSparkline(id, vh, color); } catch(e) { addLog(`📈 ${id} 圖表錯誤`); } return price; }
